pub async fn serve_grpc(
    config: &crate::config::Config,
    svc: MnemonicGrpcService,
) -> anyhow::Result<()> {
    let addr: std::net::SocketAddr = format!("0.0.0.0:{}", config.grpc_port).parse()?;

    // tonic-health: register MnemonicService as SERVING
    let (mut health_reporter, health_service) = tonic_health::server::health_reporter();
    health_reporter
        .set_serving::<MnemonicServiceServer<MnemonicGrpcService>>()
        .await;

    tracing::info!(address = %addr, "gRPC server listening");

    tonic::transport::Server::builder()
        .add_service(health_service)
        .add_service(MnemonicServiceServer::new(svc))
        .serve(addr)
        .await?;

    Ok(())
}
```

    let state = server::AppState {
        service: service.clone(),
        compaction: compaction.clone(),
        key_service: key_service.clone(),
        backend_name: config.storage_provider.clone(),
    };

    #[cfg(feature = "interface-grpc")]
    {
        // Build gRPC service struct sharing the same Arc instances (per D-12)
        let grpc_svc = grpc::MnemonicGrpcService {
            memory_service: service,
            key_service: key_service,
            compaction_service: compaction,
            backend_name: config.storage_provider.clone(),
        };

        tracing::info!(
            rest_port = config.port,
            grpc_port = config.grpc_port,
            "REST listening on 0.0.0.0:{}, gRPC listening on 0.0.0.0:{}",
            config.port,
            config.grpc_port
        );

        // Dual-port startup: first error propagates, both shut down (per D-04, D-06)
        let rest_fut = server::serve(&config, state);
        let grpc_fut = grpc::serve_grpc(&config, grpc_svc);
        tokio::try_join!(rest_fut, grpc_fut)?;
    }

    #[cfg(not(feature = "interface-grpc"))]
    {
        server::serve(&config, state).await?;
    }
```

NOTE: The `service`, `compaction`, `key_service` variables are already `Arc<_>` types from the existing code. The `.clone()` on Arc is cheap (reference count increment only). The gRPC service struct receives the original Arc while AppState gets clones (or vice versa -- both patterns are equivalent since Arc::clone is just refcount).

IMPORTANT: Make sure the variables `service`, `compaction`, `key_service` are NOT moved into AppState before the gRPC block needs them. Either clone them for AppState, or restructure so the gRPC service struct is built first and AppState uses the clones. The key constraint: both AppState and MnemonicGrpcService must hold Arc references to the same underlying service instances.
