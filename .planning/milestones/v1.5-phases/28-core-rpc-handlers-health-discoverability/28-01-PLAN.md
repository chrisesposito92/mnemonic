```rust
fn api_error_to_status(e: crate::error::ApiError) -> tonic::Status {
    use crate::error::ApiError;
    match e {
        ApiError::BadRequest(msg) => tonic::Status::invalid_argument(msg),
        ApiError::NotFound => tonic::Status::not_found("not found"),
        ApiError::Forbidden(msg) => tonic::Status::permission_denied(msg),
        ApiError::Unauthorized(msg) => tonic::Status::unauthenticated(msg),
        ApiError::Internal(e) => {
            tracing::error!(error = %e, "gRPC internal error");
            tonic::Status::internal("internal server error")
        }
    }
}
```

```rust
async fn search_memories(
    &self,
    request: tonic::Request<proto::SearchMemoriesRequest>,
) -> Result<tonic::Response<proto::SearchMemoriesResponse>, tonic::Status> {
    let auth_ctx = request.extensions().get::<crate::auth::AuthContext>().cloned();
    let body = request.into_inner();

    if body.query.trim().is_empty() {
        return Err(tonic::Status::invalid_argument("query must not be empty"));
    }

    let effective_agent_id = enforce_scope(
        auth_ctx.as_ref(),
        Some(body.agent_id.as_str()).filter(|s| !s.is_empty()),
    )
    .map_err(api_error_to_status)?;

    let mut params = crate::service::SearchParams {
        q: Some(body.query),
        agent_id: effective_agent_id
            .or_else(|| if body.agent_id.is_empty() { None } else { Some(body.agent_id) }),
        session_id: if body.session_id.is_empty() { None } else { Some(body.session_id) },
        tag: None,
        limit: if body.limit > 0 { Some(body.limit as u32) } else { None },
        threshold: None,
        after: None,
        before: None,
    };
    // Proto repeated tags: SearchMemoriesRequest has repeated string tags.
    // REST SearchParams uses `tag: Option<String>` (single tag filter).
    // Pass first tag if provided.
    if let Some(first_tag) = body.tags.first() {
        params.tag = Some(first_tag.clone());
    }

    let response = self
        .memory_service
        .search_memories(params)
        .await
        .map_err(api_error_to_status)?;

    let results = response
        .memories
        .into_iter()
        .map(|item| proto::SearchResult {
            memory: Some(memory_to_proto(item.memory)),
            distance: item.distance as f32, // f64 -> f32 cast (Pitfall 5)
        })
        .collect();

    Ok(tonic::Response::new(proto::SearchMemoriesResponse { results }))
}
```

```rust
async fn delete_memory(
    &self,
    request: tonic::Request<proto::DeleteMemoryRequest>,
) -> Result<tonic::Response<proto::DeleteMemoryResponse>, tonic::Status> {
    let auth_ctx = request.extensions().get::<crate::auth::AuthContext>().cloned();
    let body = request.into_inner();

    if body.id.is_empty() {
        return Err(tonic::Status::invalid_argument("id must not be empty"));
    }

    // D-08: Scope enforcement for scoped keys -- verify memory ownership via DB lookup
    if let Some(ref ctx) = auth_ctx {
        if let Some(ref allowed_id) = ctx.allowed_agent_id {
            match self
                .memory_service
                .get_memory_agent_id(&body.id)
                .await
                .map_err(api_error_to_status)?
            {
                None => return Err(tonic::Status::not_found("not found")),
                Some(ref mem_agent_id) if mem_agent_id != allowed_id => {
                    return Err(tonic::Status::permission_denied(format!(
                        "key scoped to {} cannot access {}",
                        allowed_id, mem_agent_id
                    )));
                }
                Some(_) => {} // ownership matches, proceed
            }
        }
    }

    let memory = self
        .memory_service
        .delete_memory(body.id)
        .await
        .map_err(api_error_to_status)?;

    Ok(tonic::Response::new(proto::DeleteMemoryResponse {
        memory: Some(memory_to_proto(memory)),
    }))
}
```
