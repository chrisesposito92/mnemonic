```rust
pub struct MemoryService {
    pub db: Arc<Connection>,
    pub embedding: Arc<dyn EmbeddingEngine>,
    pub embedding_model: String,
}

impl MemoryService {
    pub fn new(db: Arc<Connection>, embedding: Arc<dyn EmbeddingEngine>, embedding_model: String) -> Self;
    pub async fn create_memory(&self, req: CreateMemoryRequest) -> Result<Memory, ApiError>;
    pub async fn search_memories(&self, params: SearchParams) -> Result<SearchResponse, ApiError>;
    pub async fn list_memories(&self, params: ListParams) -> Result<ListResponse, ApiError>;
    pub async fn delete_memory(&self, id: String) -> Result<Memory, ApiError>;
}

pub struct CreateMemoryRequest {
    pub content: String,
    pub agent_id: Option<String>,
    pub session_id: Option<String>,
    pub tags: Option<Vec<String>>,
}

pub struct SearchParams {
    pub q: String,
    pub agent_id: Option<String>,
    pub session_id: Option<String>,
    pub tag: Option<String>,
    pub limit: Option<u32>,
    pub threshold: Option<f32>,
    pub after: Option<String>,
    pub before: Option<String>,
}

pub struct ListParams {
    pub agent_id: Option<String>,
    pub session_id: Option<String>,
    pub tag: Option<String>,
    pub after: Option<String>,
    pub before: Option<String>,
    pub limit: Option<u32>,
    pub offset: Option<u32>,
}

pub struct Memory { id, content, agent_id, session_id, tags, embedding_model, created_at, updated_at }
pub struct SearchResponse { memories: Vec<SearchResultItem> }
pub struct SearchResultItem { memory: Memory (flattened), distance: f64 }
pub struct ListResponse { memories: Vec<Memory>, total: u64 }
```
