**1. Create dashboard/src/api.ts** — typed fetch wrapper with token threading (per D-15, Research Pattern 4):

```typescript
const API_TIMEOUT_MS = 10_000

/**
 * Fetch wrapper that conditionally adds Authorization header and a 10s timeout.
 * Token is never written to localStorage — only passed from Preact component state (D-15).
 */
export function apiFetch(
  url: string,
  token: string | null,
  init?: RequestInit,
): Promise<Response> {
  const headers: Record<string, string> = {}
  if (init?.headers) {
    Object.assign(headers, init.headers)
  }
  if (token) {
    headers['Authorization'] = `Bearer ${token}`
  }
  return fetch(url, {
    ...init,
    headers,
    signal: AbortSignal.timeout(API_TIMEOUT_MS),
  })
}

/**
 * Relative time formatting (no library needed — developer tool).
 * Only needs seconds/minutes/hours/days granularity.
 */
export function relativeTime(iso: string): string {
  const diff = Math.floor((Date.now() - new Date(iso).getTime()) / 1000)
  if (diff < 0) return 'just now'
  if (diff < 60) return `${diff}s ago`
  if (diff < 3600) return `${Math.floor(diff / 60)}m ago`
  if (diff < 86400) return `${Math.floor(diff / 3600)}h ago`
  return `${Math.floor(diff / 86400)}d ago`
}

/**
 * Truncate content to a max length, appending "..." if truncated.
 * Used for memory content previews in table rows.
 */
export function truncate(text: string, maxLen: number = 80): string {
  if (text.length <= maxLen) return text
  return text.slice(0, maxLen) + '...'
}
```

```typescript
import { useState, useEffect } from 'preact/hooks'
import { apiFetch } from './api'
import Header from './components/Header'
import TabBar from './components/TabBar'
import type { Tab } from './components/TabBar'
import LoginScreen from './components/LoginScreen'
import SkeletonRows from './components/SkeletonRows'

type AppState =
  | { kind: 'checking' }
  | { kind: 'login'; error?: string }
  | { kind: 'dashboard'; token: string | null }

function parseHash(): Tab {
  const hash = window.location.hash
  if (hash === '#/agents') return 'agents'
  if (hash === '#/search') return 'search'
  return 'memories' // default
}

export default function App() {
  const [appState, setAppState] = useState<AppState>({ kind: 'checking' })
  const [activeTab, setActiveTab] = useState<Tab>(parseHash())

  // Auth detection: probe GET /health on mount (D-13)
  useEffect(() => {
    apiFetch('/health', null)
      .then(res => {
        if (res.status === 401) {
          setAppState({ kind: 'login' })
          return
        }
        if (!res.ok) throw new Error(`HTTP ${res.status}`)
        setAppState({ kind: 'dashboard', token: null })
      })
      .catch(() => {
        setAppState({ kind: 'login', error: 'Could not reach API. Check that the server is running and reload the page.' })
      })
  }, [])

  // Hash router (D-04): ~20 lines, manual useState + hashchange listener
  useEffect(() => {
    const handler = () => setActiveTab(parseHash())
    window.addEventListener('hashchange', handler)
    return () => window.removeEventListener('hashchange', handler)
  }, [])

  const handleLogin = (token: string) => {
    setAppState({ kind: 'dashboard', token })
  }

  // Render shell
  return (
    <div style={{
      minHeight: '100vh',
      background: 'var(--color-bg)',
      fontFamily: 'ui-monospace, SFMono-Regular, Menlo, Monaco, Consolas, monospace',
      color: 'var(--color-text)',
    }}>
      {appState.kind === 'checking' && (
        <div style={{ padding: '48px 24px', maxWidth: '600px', margin: '0 auto' }}>
          <SkeletonRows />
        </div>
      )}

      {appState.kind === 'login' && (
        <LoginScreen onLogin={handleLogin} />
      )}

      {appState.kind === 'dashboard' && (
        <>
          <Header token={appState.token} />
          <TabBar activeTab={activeTab} onTabChange={setActiveTab} />
          <div style={{ padding: '32px 24px', maxWidth: '1200px', margin: '0 auto' }}>
            {activeTab === 'memories' && <div>Memories tab (Plan 03)</div>}
            {activeTab === 'agents' && <div>Agents tab (Plan 04)</div>}
            {activeTab === 'search' && <div>Search tab (Plan 04)</div>}
          </div>
        </>
      )}
    </div>
  )
}
```

The tab content area renders placeholder text for now — Plans 03 and 04 will replace these with real components. The imports for those components will be added in those plans.
