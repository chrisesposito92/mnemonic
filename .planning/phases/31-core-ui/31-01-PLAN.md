```rust
async fn stats(&self) -> Result<Vec<AgentStats>, ApiError> {
    self.db.call(|c| {
        let mut stmt = c.prepare(
            "SELECT agent_id, COUNT(*) AS memory_count, MAX(created_at) AS last_active
             FROM memories
             GROUP BY agent_id
             ORDER BY last_active DESC"
        )?;
        let rows = stmt.query_map([], |row| {
            Ok(AgentStats {
                agent_id: row.get(0)?,
                memory_count: row.get(1)?,
                last_active: row.get(2)?,
            })
        })?;
        let mut agents = Vec::new();
        for row in rows {
            agents.push(row?);
        }
        Ok(agents)
    }).await.map_err(|e| ApiError::Internal(MnemonicError::Db(DbError::Query(e.to_string()))))
}
```

Add `stats()` implementation to `impl StorageBackend for QdrantBackend`. Use scroll with large limit (10_000) to fetch all points with payload (no vectors), then aggregate client-side with a HashMap. This matches the existing scroll pattern used in `list()`:

```rust
async fn stats(&self) -> Result<Vec<AgentStats>, ApiError> {
    // Scroll all points — with_payload for agent_id and created_at, no vectors needed
    let scroll = ScrollPointsBuilder::new(&self.collection)
        .limit(10_000)
        .with_payload(true)
        .with_vectors(false);

    let result = self.client.scroll(scroll)
        .await
        .map_err(|e| ApiError::Internal(MnemonicError::Db(DbError::Query(e.to_string()))))?;

    // Aggregate: HashMap<agent_id, (count, max_created_at)>
    let mut map: HashMap<String, (u64, String)> = HashMap::new();
    for point in &result.result {
        let agent_id = point.payload.get("agent_id")
            .and_then(|v| v.as_str())
            .unwrap_or("")
            .to_string();
        let created_at = point.payload.get("created_at")
            .and_then(|v| v.as_str())
            .unwrap_or("")
            .to_string();
        let entry = map.entry(agent_id).or_insert((0, String::new()));
        entry.0 += 1;
        if created_at > entry.1 {
            entry.1 = created_at.clone();
        }
    }

    let mut agents: Vec<AgentStats> = map.into_iter()
        .map(|(agent_id, (memory_count, last_active))| AgentStats {
            agent_id,
            memory_count,
            last_active,
        })
        .collect();
    agents.sort_by(|a, b| b.last_active.cmp(&a.last_active));
    Ok(agents)
}
```

```rust
async fn stats(&self) -> Result<Vec<AgentStats>, ApiError> {
    let rows = sqlx::query(
        "SELECT agent_id, COUNT(*) AS memory_count, MAX(created_at) AS last_active
         FROM memories
         GROUP BY agent_id
         ORDER BY last_active DESC"

    let agents = rows.iter().map(|row| {
        AgentStats {
            agent_id: row.get::<String, _>("agent_id"),
            memory_count: row.get::<i64, _>("memory_count") as u64,
            last_active: row.get::<String, _>("last_active"),
        }
    }).collect();

    Ok(agents)
}
```

```rust
/// Injects Content-Security-Policy header on all /ui/ responses (AUTH-02).
///
/// Policy rationale: vite-plugin-singlefile inlines all JS and CSS into index.html —
/// there are no external scripts or stylesheets. 'unsafe-inline' is required for the
/// inlined script and style tags. No external connections from the dashboard (all API
/// calls go to 'self'). This is safe for a self-hosted developer tool with no
/// user-generated content.
async fn add_csp(response: Response) -> Response {
    let (mut parts, body) = response.into_parts();
    parts.headers.insert(
        axum::http::header::CONTENT_SECURITY_POLICY,
        axum::http::HeaderValue::from_static(
            "default-src 'self'; script-src 'unsafe-inline'; style-src 'unsafe-inline'"
        ),
    );
    Response::from_parts(parts, body)
}
```

Update `pub fn router()` to wrap the serve with the CSP layer:

```rust
pub fn router() -> Router {

        Some("index.html".to_owned()),
    );
```
